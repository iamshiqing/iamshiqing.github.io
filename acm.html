1.1.2. 快读

int read(){
    int x=0,f=1;
    char ch=getchar();
    while(!isdigit(ch)){
        if(ch=='-')
            f=-1;
        ch=getchar();
    }
    while(isdigit(ch)){
        x=x*10+ch-'0';
        ch=getchar();
    }
    return x*f;
}

1.2. 快写

void write(int x){
    static int sta[35];
    int top = 0;
    if(x<0){
        x=-x;
        putchar('-');
    }

    do{
        sta[top++]=x%10;
        x/=10;
    }while(x);
    while(top) 
        putchar(sta[--top]+'0'); 
}
